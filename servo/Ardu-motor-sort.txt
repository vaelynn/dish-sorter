#include <Servo.h>
#include <SoftwareSerial.h>


SoftwareSerial BT(10, 11);  // RX=10, TX=11


Servo myServo;


const int SERVO_PIN = 9;


int currentPos = 90;


void setup() {
  BT.begin(9600);
  myServo.attach(SERVO_PIN);
  myServo.write(90);
  delay(500);
}


void loop() {
  if (BT.available() > 0) {
    char input = BT.read();


    if (input == '1') {
      for (int pos = currentPos; pos <= 180; pos++) {
        myServo.write(pos);
        delay(5);
      }
      currentPos = 180;


    } else if (input == '2') {
      for (int pos = currentPos; pos >= 0; pos--) {
        myServo.write(pos);
        delay(5);
      }
      currentPos = 0;


    } else if (input == '3') {
      if (currentPos < 90) {
        for (int pos = currentPos; pos <= 90; pos++) {
          myServo.write(pos);
          delay(5);
        }
      } else {
        for (int pos = currentPos; pos >= 90; pos--) {
          myServo.write(pos);
          delay(5);
        }
      }
      currentPos = 90;
    }
  }
}
